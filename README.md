### Bouncing Ball

    ball = new Sprite
    ball.size = 25
    ball.dx = -1
    ball.xAlign = LEFT
    ball.yAlign = BOTTOM
    on ball.x > screen.width - ball.size:
      ball.dx = -10.0
      ball.rotation = -90.0
    end
    on ball.y < 0.0:
      ball.dy = 100.0
      ball.face = "ðŸ˜²"
      wait 0.2
      ball.face = "ðŸ˜„"
    end
    on ball.x < 0.0:
      ball.dx = 10.0
      ball.rotation = 90.0
    end
    onchange screen.frame:
      ball.dy = ball.dy - 1.5
    end
